```HTML

  <link rel="canonical" href="..."/>
  <meta http-equiv="refresh" content="0; url=..."/>
<b:else/>
  <b:if cond='data:blog.url == "..."'>
    <link rel="canonical" href="..."/>
    <meta http-equiv="refresh" content="0; url=..."/>
  </b:if>
</b:if>
```
